views: from django.shortcuts import render, get_object_or_404, redirect
from .models import CamerasProfile
from .forms import CamerasProfileForm

def list_cameras(request):
    cameras = CamerasProfile.objects.all()
    return render(request, 'list_cameras.html', {'cameras': cameras})

def camera_detail(request, camera_id):
    camera = get_object_or_404(CamerasProfile, id=camera_id)
    return render(request, 'camera_detail.html', {'camera': camera})

def add_camera(request):
    if request.method == "POST":
        form = CamerasProfileForm(request.POST)
        if form.is_valid():
            form.save()
            return redirect('list_cameras')  # Supõe que você tem uma URL nomeada para a view list_cameras
    else:
        form = CamerasProfileForm()
    return render(request, 'add_camera.html', {'form': form})

models: from django.db import models

class CamerasProfile(models.Model):
    numero= models.CharField(max_length=2)
    area = models.CharField(max_length=50)  
    url = models.CharField(max_length=500)    

    class Meta:
        db_table = 'registro_cameras'

forms: from django import forms
from .models import CamerasProfile

class CamerasProfileForm(forms.ModelForm):
    class Meta:
        model = CamerasProfile
        fields = ['numero', 'area', 'url']

        function changeGif(gifSrc) {
            var mainGif = document.getElementById('mainGif');
            var thumbnails = document.querySelectorAll('.thumbnail img');
            var backButton = document.querySelector('.back-button');

            // Oculta os botões "Voltar ao Grid" e "Retornar ao Menu"
            backButton.style.display = 'block';

            // Oculta todas as miniaturas
            for (var i = 0; i < thumbnails.length; i++) {
                thumbnails[i].style.display = 'none';
            }

            // Exibe a imagem principal com o src correto
            mainGif.src = gifSrc;
            mainGif.classList.add('show');
        }

        function showThumbnails() {
            var mainGif = document.getElementById('mainGif');
            var thumbnails = document.querySelectorAll('.thumbnail img');
            var backButton = document.querySelector('.back-button');

            // Oculta a imagem principal
            mainGif.classList.remove('show');

            // Exibe todas as miniaturas
            for (var i = 0; i < thumbnails.length; i++) {
                thumbnails[i].style.display = 'block';
            }

            // Oculta o botão "Voltar ao Grid"
            backButton.style.display = 'none';
        }
